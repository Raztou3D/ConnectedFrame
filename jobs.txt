# This is to set the correct timezone on the Host
CRON_TZ=Europe/Zurich

# Turn off display every evening 
0 CRON_OFF * * * echo 1 > /sys/class/backlight/rpi_backlight/bl_power

# Turn on display every morning 
0 CRON_ON * * * echo 0 > /sys/class/backlight/rpi_backlight/bl_power

# Lowest brightness from 6PM
0 CRON_LOW * * * echo 40 > /sys/class/backlight/rpi_backlight/brightness

# Low brightness from 10AM
0 CRON_HIGH * * * echo 80 > /sys/class/backlight/rpi_backlight/brightness
